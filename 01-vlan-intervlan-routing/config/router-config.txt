! Router Configuration for Inter-VLAN Routing
enable
configure terminal
hostname R1

! Configure Subinterfaces for each VLAN
interface GigabitEthernet0/0/0
 description MAIN-TRUNK-INTERFACE
 no shutdown

interface GigabitEthernet0/0/0.10
 description VLAN10-SALES
 encapsulation dot1Q 10
 ip address 192.168.10.1 255.255.255.0

interface GigabitEthernet0/0/0.20
 description VLAN20-ENGINEERING
 encapsulation dot1Q 20
 ip address 192.168.20.1 255.255.255.0

interface GigabitEthernet0/0/0.30
 description VLAN30-IT
 encapsulation dot1Q 30
 ip address 192.168.30.1 255.255.255.0

interface GigabitEthernet0/0/0.99
 description NATIVE-VLAN-MANAGEMENT
 encapsulation dot1Q 99 native
 ip address 192.168.99.1 255.255.255.0

! Configure DHCP for all VLANs
ip dhcp excluded-address 192.168.10.1 192.168.10.10
ip dhcp excluded-address 192.168.20.1 192.168.20.10
ip dhcp excluded-address 192.168.30.1 192.168.30.10
ip dhcp excluded-address 192.168.99.1 192.168.99.10

ip dhcp pool VLAN10_POOL
 network 192.168.10.0 255.255.255.0
 default-router 192.168.10.1
 dns-server 8.8.8.8
 domain-name sales.local

ip dhcp pool VLAN20_POOL
 network 192.168.20.0 255.255.255.0
 default-router 192.168.20.1
 dns-server 8.8.8.8
 domain-name engineering.local

ip dhcp pool VLAN30_POOL
 network 192.168.30.0 255.255.255.0
 default-router 192.168.30.1
 dns-server 8.8.8.8
 domain-name it.local

ip dhcp pool VLAN99_POOL
 network 192.168.99.0 255.255.255.0
 default-router 192.168.99.1
 dns-server 8.8.8.8
 domain-name management.local

! Enable routing
ip routing

! Save Configuration
end
write memory

! Verification Commands
show ip interface brief
show ip route
show ip dhcp binding
show vlans  ! Shows subinterfaces and VLAN info
EOF
